function myAddEvent(obj, sEvent, fn){
	if(obj.attachEvent){

}

function getStyle(obj, attr){

}
function doMove(obj, json, fnEnd){

	var attr = '';

		}
	}
	if(bStop){

		obj.timer = null;		
		if(fnEnd)fnEnd();
	}
}
